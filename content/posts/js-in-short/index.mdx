#### Array

```js
// Array are maps not tuples
// 1. Array has hole
const arr = []
arr[1] = "1" // arr => [, "1"]
// 2. Array has props
arr.foo = 123


// ---------------------
// Array indices
// Indices that are out of range are treated as property key
arr[-1] = 'a';
arr[4294967296] = 'b';
// arr => []
// arr['-1'] => 'a'
// arr['4294967296'] => 'b'

// ---------------------
// Delete array elements will create hole
const arr = ['a', 'b']
delete arr[1]
// arr => ['a', ]

// So should use splice(Destructive) to not create hole
arr.splice(1,1)
// arr => ['a']

// ---------------------
// Methods that check/skip holes
// Check
.join, sort

// Skip
.forEach, every/some, map, filter, for in
// Note: `in` operator will loop through property keys and indices
const arr = [1];
arr.foo = 'bar';
for (const key in arr) {
  console.log(key); // 1, foo
}

// ---------------------
// Array prototype methods

// Add/Remove elements(Destructive)
arr.shift/unshift/pop/push/splice()
// Sorting/Reversing elements(Destructive)
arr.reverse/sort()

// Concatenating/Slicing/Joining elements(NonDestructive)
arr.concat/slice/join()
// Search(NonDestructive)
arr.indexOf/lastIndexOf()
// Iteration(NonDestructive)
arr.forEach/every/some()

// Transformation(NonDestructive)
arr.map/filter()
// Reduction(NonDestructive)
arr.reduce()
```

  const element = arr[i];

s.indexOf(",");

#### Number

```js
// Check NaN
Number.isNaN(x)

// Converting
Number(undefined) => NaN
Number(null) => 0
Number(NaN) => NaN
Number(false) => 0
Number('3.14') => 3.14
Number(o: Object) => Call ToPrimitive
```

#### Boolean

```js
// Converting
Boolean(null, undefined, NaN, 0, ""); // => false
Boolean({}); // => true for any object
```

#### Occurences of null

```js
document.getElementById("foo"); // null
```

#### Occurences of NaN

```js
// When converting variable to number
0 / 0; // NaN
Infinity - Infinity; // NaN
Math.sqrt(-1); // NaN
parseInt("foo"); // NaN
```

#### Occurences of undefined

```js
// Uninit variable
let a; // undefined

// Missing params
function foo(a, b) {
  console.log(b); // undefined
}

// Object access nonexistence prop
const o = {};
console.log(o.a); // undefined

// Function with no return
function foo() {}
console.log(foo()); // undefined
```
